SELECT instalacao.nome as Instalacao,emissao.poluente as Poluente,emissao_fact.quantidade_limiar as SUM_quantidade_limiar FROM emissao_fact,instalacao,emissao WHERE emissao_fact.instalacao_id=instalacao.instalacao_id AND emissao_fact.emissao_id=emissao.emissao_id AND (emissao.poluente='SOx' OR emissao.poluente='NOx' OR emissao.poluente='CO' OR emissao.poluente='CO2') ORDER BY instalacao.nome,emissao.poluente
